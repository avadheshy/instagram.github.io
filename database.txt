CREATE DATABASE insta;
CREATE TABLE User(
    id int NOT NULL,
    first_name Varchar(50),
    last_name Varchar(50),
    mobile int,
    email Varchar(50),
    age int,
    gender Varchar(10),
    bio Varchar(200),
    profile_pic img,
    followers_count int,
    following_count int, 
    PRIMARY KEY(id),
);
CREATE TABLE Post(
    user_id int,
    post_id int,
    post_imp img,
    post_vdo video,
    total_likes=int,
    FOREIGN KEY(post_id) REFERENCES User(id),
);
CREATE TABLE Follower(
    user_id int,
    follower_id int,
    created_at date,
    updated_at date,
    FOREIGN KEY(user_id) REFERENCES User(id),
    FOREIGN KEY(follower_id) REFERENCES User(id),

);
CREATE TABLE Reaction(
    post_id int,
    user_id int,
    created_at date,
    updated_at date,
    like_count int,
    comment Varchar(200),
    FOREIGN KEY(user_id) REFERENCES User(id),
    FOREIGN KEY(post_id) REFERENCES Post(id),

);
