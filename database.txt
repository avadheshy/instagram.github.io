CREATE DATABASE insta;
CREATE TABLE user(
    id int NOT NULL,
    first_name Varchar(50),
    last_name Varchar(50),
    mobile int,
    email Varchar(50),
    age int,
    gender Varchar(10),
    bio Varchar(200),
    profile_pic img,
    followers_count int,
    following_count int, 
    PRIMARY KEY(emp_id),
);
CREATE TABLE post(
    user_id int,
    post_id int,
    image img(),
    video video(),
    comment Varchar(200),
    date date(),
    time time(),
    like int(),
    PRIMARY KEY(post_id),
    FOREIGN KEY(post_id) REFERENCES user(id),
);
