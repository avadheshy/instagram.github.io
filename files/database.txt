CREATE DATABASE instagram;
CREATE TABLE User(
    id int,
    first_name Varchar(50),
    last_name Varchar(50),
    mobile Varchar(10),
    email Varchar(50),
    age int,
    gender Varchar(10),
    bio Varchar(200),
    profile_pic img,
    PRIMARY KEY(id),
);
CREATE TABLE Post(
    user_id int,
    post_id int,
    post_imp img,
    post_vdo video,
    last_name int,
    PRIMARY KEY(post_id),
    FOREIGN KEY(user_id) REFERENCES User(id),
);
CREATE TABLE Follower(
    user_id int,
    follower_id int,
    created_at date,
    updated_at date,
    FOREIGN KEY(user_id) REFERENCES User(id),
    FOREIGN KEY(follower_id) REFERENCES User(id),

);
CREATE TABLE Like(
    post_id int,
    user_id int,
    like_id int,
    created_at date,
    updated_at date,
    PRIMARY KEY(like_id),
    FOREIGN KEY(user_id) REFERENCES User(id),
    FOREIGN KEY(post_id) REFERENCES Post(id),

);
CREATE TABLE Comment(
    post_id int,
    user_id int,
    comment_id int,
    created_at date,
    updated_at date,
    comment Varchar(200),
    PRIMARY KEY(comment_id),
    FOREIGN KEY(user_id) REFERENCES User(id),
    FOREIGN KEY(post_id) REFERENCES Post(id),

);
